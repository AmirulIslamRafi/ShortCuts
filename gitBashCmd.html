$ git add index.html
$ rm -rf .git --//remove folder (rm -rf) .git remove all .git history
$ git diff --// its compare the working directory to staging area "if we modify a staged file"
$ git diff --staged --// its compare the previous commit to staging area.
$ git commit -a -m "Direct commit" --// directly commit the modified tracked files.
$ git add . --// adding to to the staging
$ git rm third.txt --// to remove file from git bash.
$ git mv third.txt renameThird.txt --// rename any file in git bash
$ git rm --cached third.txt --// untrack a file
$ git log --// showing all the commits and dates
$ git log -p --// showing all commit with all the detail information
$ git log -p -3 --// showing the n (3) number of commits with all the details
$ git --stat --// showing all the commits with short and important infomation
$ git log --pretty=oneline --// showing all the commits and each commi in one line
$ git log --pretty=short
$ git log --pretty=full
$ git log --since=200.days
--// showing commits information according time
$ git log --since=23.weeks
$ git log --since=2.months
$ git log --since=1.year
$ git log --prettyformat:"%h" -- %an --%ae --// %h = "hash code"; %an ="author name"; %ae = "author email"
$ git commit --amend --// edit the recent commit
$ git restore --staged jabed.html $ git checkout -- jabed.html or $ git restore jabed.html --// return back from modified to last commit
$ git checkout -f --// (all modified files) return back from modified to last commit
$ git config --global alias.putki status --// with alias we can rename command line keyword
$ git config --global alias.unstage 'restore --staged'
$ git remote
$ git remote -v
$ git branch -m OLD-BRANCH-NAME NEW-BRANCH-NAME --// change the exsisting branch name
$ git checkout -b develop --// creating and switching in develop (NEW-BRANCH)
$ git checkout master --// switching to the master branch
